  const [selected, setSelected] = useState(null);
  const [score, setScore] = useState(0);
  const [done, setDone] = useState(false);

  const q = category.questions[idx];
  const c = q[lang];

  function choose(i) {
    if (selected !== null) return;
    setSelected(i);
    if (i === q.answer) setScore((s) => s + 1);
  }

  function next() {
    if (idx + 1 < category.questions.length) {
      setIdx(idx + 1);
      setSelected(null);
    } else {
      setDone(true);
    }
  }

  if (done) {
    const pct = Math.round((score / category.questions.length) * 100);
    return (
      <div className="p-6 flex flex-col items-center text-center pb-10">
        <div className="w-20 h-20 rounded-full flex items-center justify-center mb-4" style={{ background: `${category.color}18` }}>
          <Award size={34} color={category.color} />
        </div>
        <div className="text-[13px] text-slate-500 mb-1">{t.yourScore}</div>
        <div className="text-3xl font-extrabold text-slate-800 mb-1">
          {score} / {category.questions.length}
        </div>
        <div className="text-[13px] font-semibold mb-6" style={{ color: category.color }}>{pct}%</div>
        <div className="flex gap-2 w-full max-w-xs">
          <button
            onClick={() => { setIdx(0); setSelected(null); setScore(0); setDone(false); }}
            className="flex-1 flex items-center justify-center gap-1.5 bg-slate-100 text-slate-600 font-semibold text-[13px] py-2.5 rounded-xl"
          >
            <RotateCcw size={14} /> {t.tryAgain}
          </button>
          <button
            onClick={onExit}
            className="flex-1 bg-[#0B2F5E] text-white font-semibold text-[13px] py-2.5 rounded-xl"
          >
            {t.backToQuiz}
          </button>
        </div>
      </div>
    );
  }

  return (
    <div className="p-4 pb-6">
      <div className="flex items-center justify-between mb-3">
        <span className="text-[12px] font-semibold text-slate-400">
          {idx + 1} {t.of} {category.questions.length}
        </span>
        <span className="text-[12px] font-bold" style={{ color: category.color }}>{category[lang].name}</span>
      </div>
      <div className="h-1.5 bg-slate-100 rounded-full mb-5 overflow-hidden">
        <div
          className="h-full rounded-full transition-all"
          style={{ width: `${((idx + (selected !== null ? 1 : 0)) / category.questions.length) * 100}%`, background: category.color }}
        />
      </div>
      <h2 className="font-bold text-slate-800 text-[15.5px] mb-4 leading-snug">{c.q}</h2>
      <div className="space-y-2.5 mb-5">
        {c.options.map((opt, i) => {
          const isAnswer = i === q.answer;
          const isChosen = i === selected;
          let style = "bg-white border-slate-200 text-slate-700";
          if (selected !== null) {
            if (isAnswer) style = "bg-emerald-50 border-emerald-400 text-emerald-700";
            else if (isChosen) style = "bg-red-50 border-red-400 text-red-600";
          }
          return (
            <button
              key={i}
              onClick={() => choose(i)}
              className={`w-full text-left border rounded-xl px-3.5 py-2.5 text-[13.5px] font-medium flex items-center justify-between ${style}`}
            >
              {opt}
              {selected !== null && isAnswer && <CheckCircle2 size={17} className="text-emerald-500 shrink-0" />}
              {selected !== null && isChosen && !isAnswer && <XCircle size={17} className="text-red-500 shrink-0" />}
            </button>
          );
        })}
      </div>
      {selected !== null && (
        <button
          onClick={next}
          className="w-full text-white font-semibold text-[13.5px] py-2.5 rounded-xl"
          style={{ background: category.color }}
        >
          {idx + 1 < category.questions.length ? t.next : t.finish}
        </button>
      )}
    </div>
  );
}

export default function AdduElectricalLearning() {
  const [lang, setLang] = useState("am");
  const [screen, setScreen] = useState("home");
  const [deviceId, setDeviceId] = useState(null);
  const [quizCatId, setQuizCatId] = useState(null);

  const t = UI[lang];

  const titles = {
    home: t.appName,
    basics: t.basicElectricity,
    devices: t.electricalDevices,
    device: DEVICES.find((d) => d.id === deviceId)?.name[lang] ?? "",
    diagrams: t.circuitDiagrams,
    safety: t.electricalSafety,
    about: t.about,
    quiz: t.quiz,
    quizRun: QUIZ_CATEGORIES.find((c) => c.id === quizCatId)?.[lang]?.name ?? "",
  };

  const backable = screen !== "home";
  const showBottomNav = ["home", "basics", "devices", "diagrams", "quiz"].includes(screen);

  function handleBack() {
    if (screen === "device") setScreen("devices");
    else if (screen === "quizRun") setScreen("quiz");
    else setScreen("home");
  }

  return (
    <div
      className="mx-auto w-full max-w-md min-h-screen flex flex-col bg-[#F5F7FB]"
      dir="ltr"
      style={{ fontFamily: "'Segoe UI', system-ui, sans-serif" }}
    >
      <TopBar
        t={t}
        title={titles[screen] || t.appName}
        onBack={backable ? handleBack : null}
        lang={lang}
        setLang={setLang}
      />

      <div className="flex-1 overflow-y-auto">
        {screen === "home" && <HomeScreen t={t} lang={lang} setScreen={setScreen} />}

        {screen === "basics" && (
          <>
            <BasicsScreen t={t} lang={lang} />
          </>
        )}

        {screen === "devices" && (
          <DevicesScreen
            t={t}
            lang={lang}
            onOpen={(id) => {
              setDeviceId(id);
              setScreen("device");
            }}
          />
        )}

        {screen === "device" && deviceId && (
          <DeviceDetailScreen t={t} lang={lang} device={DEVICES.find((d) => d.id === deviceId)} />
        )}

        {screen === "diagrams" && <DiagramsScreen t={t} lang={lang} />}

        {screen === "safety" && <SafetyScreen t={t} lang={lang} />}

        {screen === "about" && <AboutScreen t={t} lang={lang} />}

        {screen === "quiz" && (
          <QuizMenuScreen
            t={t}
            lang={lang}
            onPick={(id) => {
              setQuizCatId(id);
              setScreen("quizRun");
            }}
          />
        )}

        {screen === "quizRun" && quizCatId && (
          <QuizRunScreen
            t={t}
            lang={lang}
            category={QUIZ_CATEGORIES.find((c) => c.id === quizCatId)}
            onExit={() => setScreen("quiz")}
          />
        )}
      </div>

      {showBottomNav && <BottomNav t={t} screen={screen} setScreen={setScreen} />}
    </div>
  );
}
